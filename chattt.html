public class ChatServer
{
    private TcpListener _tcpListener;
    private ClientManager _clientManager;
    private UserManager _userManager;
    private bool _isRunning;
    
    public void Start()
    {
        _tcpListener = new TcpListener(IPAddress.Any, 12345);
        _clientManager = new ClientManager();
        _userManager = new UserManager();
        
        _tcpListener.Start();
        _isRunning = true;
        
        Console.WriteLine("Chat Server started on port 12345...");
        
        // í´ë¼ì´ì–¸íŠ¸ ì—°ê²° ëŒ€ê¸° ë£¨í”„
        while (_isRunning)
        {
            try
            {
                TcpClient tcpClient = _tcpListener.AcceptTcpClient();
                Console.WriteLine($"New client connected: {tcpClient.Client.RemoteEndPoint}");
                
                // ê° í´ë¼ì´ì–¸íŠ¸ë¥¼ ë³„ë„ ìŠ¤ë ˆë“œì—ì„œ ì²˜ë¦¬
                ClientSession session = new ClientSession(tcpClient, _clientManager, _userManager);
                Thread clientThread = new Thread(session.HandleClient);
                clientThread.Start();
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Error accepting client: {ex.Message}");
            }
        }
    }
}

public class ClientSession
{
    private TcpClient _tcpClient;
    private NetworkStream _networkStream;
    private StreamReader _reader;
    private StreamWriter _writer;
    private ClientManager _clientManager;
    private UserManager _userManager;
    private string _username;
    
    public void HandleClient()
    {
        try
        {
            _networkStream = _tcpClient.GetStream();
            _reader = new StreamReader(_networkStream);
            _writer = new StreamWriter(_networkStream) { AutoFlush = true };
            
            string message;
            while ((message = _reader.ReadLine()) != null)
            {
                ProcessMessage(message);
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Client session error: {ex.Message}");
        }
        finally
        {
            Cleanup();
        }
    }
    
    private void ProcessMessage(string message)
    {
        string[] parts = message.Split(':');
        string command = parts[0];
        
        switch (command)
        {
            case "LOGIN":
                HandleLogin(parts[1], parts[2]);
                break;
            case "REGISTER":
                HandleRegister(parts[1], parts[2]);
                break;
            case "CHAT_MESSAGE":
                HandleChatMessage(parts[1], parts[2]);
                break;
            case "REQUEST_PARTICIPANTS":
                HandleParticipantsRequest();
                break;
        }
    }
    
    private void HandleLogin(string username, string password)
    {
        if (_userManager.ValidateUser(username, password))
        {
            if (_clientManager.IsUserLoggedIn(username))
            {
                SendMessage("LOGIN_FAIL:AlreadyLoggedIn");
                return;
            }
            
            _username = username;
            _clientManager.AddClient(this, username);
            SendMessage($"LOGIN_SUCCESS:{username}");
            
            // ë‹¤ë¥¸ í´ë¼ì´ì–¸íŠ¸ë“¤ì—ê²Œ ì…ì¥ ì•Œë¦¼
            _clientManager.BroadcastMessage($"USER_JOINED:{username}", username);
        }
        else
        {
            SendMessage("LOGIN_FAIL:InvalidUser");
        }
    }
    
    private void HandleChatMessage(string sender, string messageContent)
    {
        if (_username == sender)
        {
            // ëª¨ë“  í´ë¼ì´ì–¸íŠ¸ì—ê²Œ ë©”ì‹œì§€ ë¸Œë¡œë“œìºìŠ¤íŠ¸
            _clientManager.BroadcastMessage($"CHAT_MESSAGE:{sender}:{messageContent}");
        }
    }
}

public class ClientManager
{
    private readonly Dictionary<string, ClientSession> _clients = new();
    private readonly object _lock = new object();
    
    public void AddClient(ClientSession session, string username)
    {
        lock (_lock)
        {
            _clients[username] = session;
        }
        Console.WriteLine($"User {username} added to client manager. Total clients: {_clients.Count}");
    }
    
    public void RemoveClient(string username)
    {
        lock (_lock)
        {
            if (_clients.ContainsKey(username))
            {
                _clients.Remove(username);
                Console.WriteLine($"User {username} removed from client manager. Total clients: {_clients.Count}");
            }
        }
    }
    
    public void BroadcastMessage(string message, string excludeUser = null)
    {
        List<ClientSession> clientsToSend;
        
        lock (_lock)
        {
            clientsToSend = _clients.Values
                .Where(client => excludeUser == null || client.Username != excludeUser)
                .ToList();
        }
        
        // ë½ ì™¸ë¶€ì—ì„œ ë©”ì‹œì§€ ì „ì†¡ (ë°ë“œë½ ë°©ì§€)
        foreach (var client in clientsToSend)
        {
            try
            {
                client.SendMessage(message);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Failed to send message to client: {ex.Message}");
            }
        }
    }
    
    public bool IsUserLoggedIn(string username)
    {
        lock (_lock)
        {
            return _clients.ContainsKey(username);
        }
    }
    
    public List<string> GetParticipantsList()
    {
        lock (_lock)
        {
            return _clients.Keys.ToList();
        }
    }
}
